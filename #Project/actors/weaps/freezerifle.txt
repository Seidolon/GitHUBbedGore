actor FreezeRifle : Weapon 30123
{
//$Category New Weapons
Tag "Freeze Rifle"
Game Doom
Weapon.AmmoUse 1
Weapon.AmmoGive 25
Weapon.AmmoType "IceGlobe"
Weapon.SlotNumber 6
Inventory.PickUpSound "weapons/freezepk"
Inventory.PickupMessage "You got the Freeze rifle!"
States
	{
	Spawn:
	FRAS A -1
	Stop
	Ready:
	FRSG A 1 A_WeaponReady
	Loop
	Deselect:
	FRSG A 1 A_Lower
	Loop
	Select:
	FRSG A 1 A_Raise
	Loop
	Fire:
	FRSG A 0 A_GunFlash
	FRSG A 6 A_FireCustomMissile("FreezeBlast")
	FRSG B 20 A_ReFire
	Goto Ready
	Flash:
	FRSF A 3 Bright A_Light1
	Goto LightDone
	FRSF B 3 Bright A_Light1
	Goto LightDone
	}
}
